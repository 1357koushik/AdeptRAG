import babase
import _babase
import badwords
# Configs
LOG_FILE = "Filter_Chat_Log.txt"
ADMINS=['pb-IF4pU2JeDg==']
MAX_WARNINGS = 3


# --- Load bad words ---
def load_badwords():
    try:
        return set(badwords.words.strip().lower())
    except Exception as e:
        babase.print_error(f"[ProfanityFilter] Failed to load badwords: {e}")
        return set()

BADWORDS = load_badwords()
WORD_RE = re.compile(r'\W+')

# --- Logger ---
def log_violation(name, account_id, message, level):
    try:
        with open(LOG_FILE, 'a', encoding='utf-8') as f:
            f.write(f"[{level}] {name} ({account_id}): {message}\n")
    except Exception as e:
        babase.print_error(f"[ProfanityFilter] Failed to write log: {e}")

# --- Warning Manager Class ---
class PlayerWarningManager:
    def __init__(self):
        self._warnings = {}

    def add_warning(self, account_id):
        self._warnings.setdefault(account_id, 0)
        self._warnings[account_id] += 1
        return self._warnings[account_id]

    def get_warning_count(self, account_id):
        return self._warnings.get(account_id, 0)

    def reset(self, account_id):
        if account_id in self._warnings:
            del self._warnings[account_id]

warning_manager = PlayerWarningManager()

# --- Main Chat Filter ---
def on_chat_message(msg,client_id,account_id,name):
    try:
        msg_text = msg.lower()
        words = set(WORD_RE.sub(' ', msg_text).split())

        if account_id in ADMINS:
            return  True

        if words & BADWORDS:
            count = warning_manager.add_warning(account_id)
            log_violation(name, account_id, msg_text, f"WARNING {count}")

            if count >= MAX_WARNINGS:
                babase.screenmessage(f"\U0001F6AB {name} kicked for offensive language.", color=(1, 0, 0))
                _babase.disconnect_client(client_id)
                warning_manager.reset(account_id)
            else:
                babase.screenmessage(
                    f"\u26A0 {name}, warning {count}/{MAX_WARNINGS} for language.",
                    color=(1, 1, 0)
                )
        else:
            return True
        

    except Exception as e:
        babase.print_error(f"[ProfanityFilter] Error handling chat: {e}")
        return True
